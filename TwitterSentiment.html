def create_tweet_df(tweets):

    tweet_dict = {'created_at': [],
                  'favorite_count': [],
                  'hashtags': [],
                  'id': [],
                  'expanded_url': [],
                  'tweet': []}
    for ut in tweets:
        tweet_dict['tweet'].append((ut.text))
        tweet_dict['created_at'].append(ut.created_at)
        try:
            tweet_dict['hashtags'].append(','.join(ht.text for ht in ut.hashtags))
        except KeyError:
            tweet_dict['hashtags'].append('')
        tweet_dict['id'].append(ut.id)
        try:
            tweet_dict['expanded_url'].append(','.join(url.expanded_url for url in ut.urls))
        except KeyError:
            tweet_dict['expanded_url'].append('')
        tweet_dict['favorite_count'].append(ut.favorite_count)

    df = pd.DataFrame(tweet_dict)
    return df


def remove_links(tweet):
    link_pattern = re.compile(r"(?:\@|https?\://)\S+")
    return link_pattern.sub(r'', tweet)


def remove_emojis(tweet):
    emoji_pattern = re.compile(
                               u"(\ud83d[\ude00-\ude4f])|"  # emoticons
                               u"(\ud83c[\udf00-\uffff])|"  # symbols & pictographs (1 of 2)
                               u"(\ud83d[\u0000-\uddff])|"  # symbols & pictographs (2 of 2)
                               u"(\ud83d[\ude80-\udeff])|"  # transport & map symbols
                               u"(\ud83c[\udde0-\uddff])"  # flags (iOS)
                               "+", flags=re.UNICODE)
    return emoji_pattern.sub(r'', tweet)


def get_tweets(api, screen_name, total_count=200):
    Ntweets = total_count
    user_tweets = []
    uts = api.GetUserTimeline(screen_name=screen_name, count=200)
    user_tweets.extend(uts)
    maxid = uts[-1].id

    count = 200
    while count < Ntweets:
        uts = api.GetUserTimeline(screen_name=screen_name, count=200, max_id=maxid)
        user_tweets.extend(uts)
        maxid = uts[-1].id
        count += 200
    return user_tweets

Getting the tweets¶

Now lets get the most recent 1000 tweets from the user "realDonaldTrump". To do this we use the GetUserTimeline attribute. The API call returns a Status object which has a bunch of meta information in addition to the tweet text such as the date it was created, the hashtags used, etc.

Note that the call only returns the 200 most recent tweets so have to do some extra work to get 1000 tweets.

Above we created a function to create a pandas dataframe and we then parse the tweets to remove links and emojis.

The tweets are cleaned up we can now get the sentiment polarity and subjectivity of the tweet and do some simple analysis. This is incredibly easy using TextBlob. The polarity ranges from -1.0 to 1.0 where values less than zero have a negative connotation and values greater than zero have a positive connotation. Subjectivity ranges from 0 to 1 with 0 being very objective and 1 being very subjective.

In [6]:
tweet_frame['sentiment'] = tweet_frame['tweet'].apply(lambda x: TextBlob(x).polarity)
tweet_frame['subjectivity'] = tweet_frame['tweet'].apply(lambda x: TextBlob(x).subjectivity)
